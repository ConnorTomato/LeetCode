#### [32. 最长有效括号](https://leetcode-cn.com/problems/longest-valid-parentheses/)

> - 栈模拟
> - 执行用时：2 ms, 在所有 Java 提交中击败了64.40%的用户

```java
class Solution {
    public int longestValidParentheses(String s) {
        int len = s.length();
        if(len == 0){
            return 0;
        }
        Deque<Integer> stack = new ArrayDeque<>();
        boolean[] marked = new boolean[len];
        for(int i = 0; i < len; i++){
            if(s.charAt(i) =='('){
                stack.push(i);
            }else{
                if(stack.isEmpty()){
                    marked[i] = true;
                }else{
                   stack.pop();
                }
            }
        }
        while(!stack.isEmpty()){
            marked[stack.pop()] = true;
        }
        // 求最长连续0
        int ans = 0, size = 0;
        for(int i = 0; i < len; i++){
            if(marked[i]){
                size = 0;
                continue;
            }
            size++;
            ans = Math.max(ans, size);
        }
        return ans;
    }
}
```

```java
class Solution {
    public int longestValidParentheses(String s) {
        int len = s.length();
        if(len == 0){
            return 0;
        }
        char[] chars = s.toCharArray();
        int[] dp = new int[len];
        dp[0] = 0;
        int res = 0;
        for(int i = 1; i < len; i++){
            if(chars[i] == ')'){
                if(chars[i-1] == '('){
                    dp[i] =( i >= 2 ? dp[i-2] : 0) + 2;
                }else if(i - dp[i-1] > 0 && chars[i - dp[i-1] - 1] == '('){
                    dp[i] = dp[i-1] +( i - dp[i-1] >= 2 ? dp[i-dp[i-1] -2] : 0) + 2;
                }
            }
            res = Math.max(res,dp[i]);
        }
        return res;
    }
}
```
