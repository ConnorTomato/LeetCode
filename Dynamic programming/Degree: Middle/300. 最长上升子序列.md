import java.util.*;

public class Solution {
    public int[] LIS (int[] arr) {
        // write code here
        int len = arr.length;
        // 特判
        if(len == 0){
            return new int[0];
        }
        // 以arr[i]结尾最长子序列长度
        int[] dp = new int[len];
        // 长度为i的子序列的最后一个元素的最小值
        int[] end = new int[len + 1];
        end[0] = Integer.MIN_VALUE; // 序列中可能有负数
        end[1] = arr[0];
        // 当前找到的最长子序列的长度
        int size = 1;
        for(int i = 0; i < len; i++){
            if(end[size] < arr[i]){
                end[++size] = arr[i];
                dp[i] = size;
            }else{ // 查找第一个大于arr[i]的元素
                int l = 0, r = size;
                while(l < r){
                    int mid = (l + r) >>> 1;
                    if(end[mid] < arr[i]){
                        l = mid + 1;
                    }else{
                        r = mid;
                    }
                }
                end[l] = arr[i];
                dp[i] = l;
            }
        }
        int[] res = new int[size];
        for(int i = len - 1; i >= 0; i--){
            if(dp[i] == size){
                res[--size] = arr[i];
            }
        }
        return res;
    }
}
