#### [18. 四数之和](https://leetcode-cn.com/problems/4sum/)

> - 先定住两个数去重然后再用双指针
> - 执行用时：4 ms, 在所有 Java 提交中击败了93.20%的用户（剪枝从28 -> 93）

```java
class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        int len = nums.length;
        if(len < 4){
            return Collections.emptyList();
        }
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        Arrays.sort(nums);
        for(int i = 0; i < len - 3; i++){
            if(i > 0 && nums[i] == nums[i - 1]){
                continue;
            }
            //获取当前最小值，如果最小值比目标值大，说明后面越来越大的值直接跳出
            int minSum = nums[i] + nums[i + 1] + nums[i + 2] + nums[i + 3];
            if(minSum > target){
                break;
            }
            //获取当前最大值，如果最大值比目标值小，说明后面越来越小的值根本没戏，忽略
            int maxSum = nums[i] + nums[len - 1] + nums[len - 2] + nums[len - 3];
            if(maxSum < target){
                continue;
            }
            for(int j = i + 1; j < len - 2; j++){
                if(j > i + 1 && nums[j] == nums[j - 1]){
                    continue;
                }
                int left = j + 1;
                int right = len - 1;
                int curMin = nums[i] + nums[j] + nums[left] + nums[left + 1];
                if(curMin > target){
                    break;
                }
                int curMax = nums[i] + nums[j] + nums[right] + nums[right - 1];
                if(curMax < target){
                    continue;
                }
                while(left < right){
                    int curSum = nums[left] + nums[right] + nums[i] + nums[j];
                    if(curSum == target){
                        res.add(Arrays.asList(nums[i], nums[j], nums[left], nums[right]));
                        while(left < right && nums[left] == nums[left + 1]) left++;
                        while(left < right && nums[right] == nums[right - 1]) right--;
                        left++;
                        right--;
                    }else if(curSum < target){
                        while(left < right && nums[left] == nums[left + 1]) left++;
                        left++;
                    }else if(curSum > target){
                        while(left < right && nums[right] == nums[right - 1]) right--;
                        right--;
                    }
                }
            }
        }
        return res;
    }
}
```
