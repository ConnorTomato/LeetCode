####[698. 划分为k个相等的子集](https://leetcode.cn/problems/partition-to-k-equal-sum-subsets/)
``` java
class Solution {
    public boolean canPartitionKSubsets(int[] nums, int k) {
        int len = nums.length;
        if (len < k) {
            return false;
        }
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        if (sum % k != 0) {
            return false;
        }
        Arrays.sort(nums);
        // 存在一个数大于平均值，false
        if (nums[len - 1] > sum / k) {
            return false;
        }
        // k 个桶
        int[] bucket = new int[k];
        Arrays.fill(bucket, sum / k);
        return backTracking(nums, len - 1, bucket, k);
    }
    private boolean backTracking(int[] nums, int cur, int[] bucket, int k) {
        if (cur < 0) {
            return true;
        }
        for (int i = 0; i < k; i++) {
            if (nums[cur] == bucket[i] || bucket[i] - nums[cur] >= nums[0]) {
                bucket[i] -= nums[cur];
                if (backTracking(nums, cur - 1, bucket, k)) {
                    return true;
                }
                bucket[i] += nums[cur];
            }
        }
        return false;
    }
}
```
