class Solution {
    public int numTrees(int n) {
        int[] G = new int[n+1];
        G[0] = 1;
        G[1] = 1;
        for(int i = 2; i <= n; i++){
            for(int j = 1 ;j <=i; j++){
                G[i] += G[j-1] * G[i-j]; 
            }
        }
        return G[n];
    }
}

public int numTrees(int n) {
    // 1. 定义 dp 数组
    // dp[i] 表示由 i 个节点组成的唯一 BST 的数量
    int[] dp = new int[n + 1];

    // 2. 基础状态（边界条件）
    // 0 个节点（空树）有一种形态
    // 1 个节点有一种形态
    dp[0] = 1;

    // 3. 开始递推
    // 外层循环 i：代表当前节点总数，从 1 增长到 n
    for(int i = 1; i <= n; i++){
        // 内层循环 j：代表在 i 个节点中，选择第 j 个节点作为根节点
        for(int j = 1; j <= i; j++){
            // 状态转移方程：
            // dp[j-1] 是左子树的组合数（因为根是 j，左边有 j-1 个比它小的）
            // dp[i-j] 是右子树的组合数（总共 i 个，减去根和左边，剩下 i-j 个）
            dp[i] += dp[j - 1] * dp[i - j];
        }
    }
    
    // 4. 返回结果
    return dp[n];
}

class Solution {
    public int numTrees(int n) {
        if(n == 0 || n == 1){
            return 1;
        }
        return helper(1,n);
    }
    private int helper(int start, int end){
        if(start >= end){
            return 1;
        }
        int ans = 0;
        for(int i = start; i <= end; i++){
            int left = helper(start,i-1);
            int right = helper(i + 1, end);
            ans += left * right;
        }
        return ans;
    }
}

class Solution {
    public int numTrees(int n) {
        if(n == 0 || n == 1){
            return 1;
        }
        HashMap<Integer, Integer> memo = new HashMap<>();
        return helper(n, memo);
    }
    private int helper(int n, HashMap<Integer, Integer> memo){
        if(n == 0 || n == 1){
            return 1;
        }
        if(memo.containsKey(n)){
            return memo.get(n);
        }
        int res = 0;
        for(int i = 1; i <= n; i++){
            int left = helper(i - 1, memo);
            int right= helper(n - i, memo);
            res += left * right;
        }
        memo.put(n, res);
        return res;
    }
}
