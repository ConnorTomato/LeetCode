class Solution {
    public int orangesRotting(int[][] grid) {
        int[][] dirctions = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        Deque<int[]> queue = new ArrayDeque<>();
        int ans = 0;
        //æŠŠè…çƒ‚çš„ğŸŠåŠ å…¥é˜Ÿåˆ—ä¸­ï¼Œä½œä¸ºå¼€å§‹æ‰©æ•£çš„èµ·ç‚¹
        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == 2){
                    queue.offer(new int[]{i, j});
                }
            }
        }
        while(!queue.isEmpty()){
            int size = queue.size();
            for(int i = 0; i < size; i++){
            int[] cell = queue.poll();
            for(int[] dir : dirctions){
                int newX = cell[0] + dir[0];
                int newY = cell[1] + dir[1];
                if(inArea(newX, newY, grid) && grid[newX][newY] == 1){
                    grid[newX][newY] = 2;
                    queue.offer(new int[]{newX, newY});
                    }
                }                
            }
            // æ³¨æ„ï¼šæœ€åè™½ç„¶å…¨éƒ¨æ„ŸæŸ“ï¼Œä½†é˜Ÿåˆ—ä¸­ä¾ç„¶æœ‰æ©˜å­
            if(!queue.isEmpty()){
                ans += 1;
            }
        }
        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == '1'){
                    return -1;
                }
            }
        }
        return ans;
    }
    private boolean inArea(int x, int y, int[][] grid){
        return x >= 0 && x < grid.length && y >= 0 && y < grid[0].length;
    }
}
